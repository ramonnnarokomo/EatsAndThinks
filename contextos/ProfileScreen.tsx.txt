import { useState, useEffect } from 'react';
import { Card } from './ui/card';
import { Avatar, AvatarFallback } from './ui/avatar';
import { Badge } from './ui/badge';
import { Button } from './ui/button';
import { Input } from './ui/input';
import { Label } from './ui/label';
import { Textarea } from './ui/textarea';
import { Separator } from './ui/separator';
import { Star, Mail, Calendar, Edit, Trash2, X, Save, Trophy, TrendingUp, MapPin, Sparkles } from 'lucide-react';
import { useAuth } from '../context/AuthContext';
import api from '../services/api';
import { deleteReview } from '../services/reviews';

interface UserData {
  id: number;
  nombre: string;
  email: string;
  createdAt: string;
  totalReviews: number;
  avgRating: number;
}

interface UserReview {
  reviewId: number;
  restaurantName: string;
  puntuacion: number;
  comentario: string;
  fecha: string;
  placeId: string;
}

export function ProfileScreen({ onRestaurantClick }: { onRestaurantClick: (id: string) => void }) {
  const { user, logout } = useAuth();
  const [userData, setUserData] = useState<UserData | null>(null);
  const [reviews, setReviews] = useState<UserReview[]>([]);
  const [loading, setLoading] = useState(true);
  const [editMode, setEditMode] = useState(false);
  const [editingReview, setEditingReview] = useState<number | null>(null);
  const [editData, setEditData] = useState({ nombre: '', email: '' });
  const [editReviewData, setEditReviewData] = useState({ puntuacion: 5, comentario: '' });

  useEffect(() => {
    loadUserData();
  }, []);

  const loadUserData = async () => {
    try {
      setLoading(true);
      const [userRes, reviewsRes] = await Promise.all([
        api.get('/users/me'),
        api.get('/users/me/reviews')
      ]);
      
      setUserData(userRes.data);
      setReviews(reviewsRes.data);
      setEditData({ nombre: userRes.data.nombre, email: userRes.data.email });
    } catch (error) {
      console.error('Error cargando datos:', error);
    } finally {
      setLoading(false);
    }
  };

  const handleUpdateProfile = async () => {
    try {
      await api.put('/users/me', editData);
      await loadUserData();
      setEditMode(false);
      alert('‚úÖ Perfil actualizado correctamente');
    } catch (error: any) {
      alert('Error: ' + (error.response?.data?.message || 'No se pudo actualizar'));
    }
  };

  const handleDeleteReview = async (reviewId: number) => {
    if (!confirm('¬øEst√°s seguro de eliminar esta rese√±a?')) return;
    
    try {
      await deleteReview(reviewId);
      setReviews(reviews.filter(r => r.reviewId !== reviewId));
      alert('‚úÖ Rese√±a eliminada');
    } catch (error) {
      alert('‚ùå Error al eliminar rese√±a');
    }
  };

  const handleEditReview = (review: UserReview) => {
    setEditingReview(review.reviewId);
    setEditReviewData({ puntuacion: review.puntuacion, comentario: review.comentario });
  };

  const handleSaveReview = async (reviewId: number) => {
    try {
      await api.put(`/reviews/${reviewId}`, editReviewData);
      await loadUserData();
      setEditingReview(null);
      alert('‚úÖ Rese√±a actualizada');
    } catch (error) {
      alert('‚ùå Error al actualizar rese√±a');
    }
  };

  if (loading) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-gray-50 via-blue-50/30 to-orange-50/30 flex items-center justify-center">
        <div className="text-center">
          <div className="animate-spin rounded-full h-16 w-16 border-b-4 border-blue-600 mx-auto mb-4"></div>
          <p className="text-gray-600 text-lg">Cargando tu perfil...</p>
        </div>
      </div>
    );
  }

  if (!userData) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-gray-50 via-blue-50/30 to-orange-50/30 flex items-center justify-center">
        <Card className="p-8 text-center">
          <p className="text-gray-600 mb-4">Error al cargar el perfil</p>
          <Button onClick={loadUserData}>Reintentar</Button>
        </Card>
      </div>
    );
  }

  const memberSince = new Date(userData.createdAt).toLocaleDateString('es-ES', {
    year: 'numeric',
    month: 'long'
  });

  return (
    <div className="min-h-screen bg-gradient-to-br from-gray-50 via-blue-50/30 to-orange-50/30">
      {/* Header con gradiente animado */}
      <div className="relative bg-gradient-to-r from-blue-600 via-blue-700 to-orange-600 overflow-hidden">
        <div className="absolute inset-0 bg-black/10"></div>
        <div className="absolute inset-0">
          <div className="absolute top-0 left-1/4 w-96 h-96 bg-orange-400/20 rounded-full blur-3xl animate-pulse"></div>
          <div className="absolute bottom-0 right-1/4 w-96 h-96 bg-blue-400/20 rounded-full blur-3xl animate-pulse delay-1000"></div>
        </div>
        
        <div className="relative max-w-6xl mx-auto px-6 py-12">
          <div className="flex items-center gap-2 mb-4">
            <Sparkles className="w-6 h-6 text-yellow-300" />
            <span className="text-white/90 text-sm font-medium">Mi Perfil</span>
          </div>
          <h1 className="text-4xl font-bold text-white mb-2">
            ¬°Hola, {userData.nombre}!
          </h1>
          <p className="text-white/80 text-lg">
            Gestiona tu perfil y revisa tu actividad gastron√≥mica
          </p>
        </div>
      </div>

      <div className="max-w-6xl mx-auto px-6 py-8 -mt-8">
        {/* Tarjeta de perfil principal */}
        <Card className="p-8 rounded-3xl shadow-2xl mb-6 bg-white/80 backdrop-blur border-2 border-white">
          <div className="flex flex-col md:flex-row items-start gap-8">
            <div className="relative group">
              <div className="absolute -inset-1 bg-gradient-to-r from-blue-600 to-orange-600 rounded-full blur opacity-25 group-hover:opacity-75 transition duration-1000"></div>
              <Avatar className="relative w-32 h-32 ring-4 ring-white shadow-xl">
                <AvatarFallback className="bg-gradient-to-br from-blue-500 to-orange-500 text-white text-3xl font-bold">
                  {userData.nombre.split(' ').map(n => n[0]).join('')}
                </AvatarFallback>
              </Avatar>
            </div>
            
            <div className="flex-1">
              {!editMode ? (
                <>
                  <div className="flex items-start justify-between mb-4">
                    <div>
                      <h2 className="text-3xl font-bold text-gray-900 mb-2">{userData.nombre}</h2>
                      <div className="flex items-center gap-2 text-gray-600 mb-2">
                        <Mail className="w-4 h-4" />
                        <span>{userData.email}</span>
                      </div>
                      <div className="flex items-center gap-2 text-gray-600">
                        <Calendar className="w-4 h-4" />
                        <span>Miembro desde {memberSince}</span>
                      </div>
                    </div>
                    <Button 
                      onClick={() => setEditMode(true)}
                      className="bg-gradient-to-r from-blue-600 to-blue-700 hover:from-blue-700 hover:to-blue-800 rounded-xl shadow-lg"
                    >
                      <Edit className="w-4 h-4 mr-2" />
                      Editar perfil
                    </Button>
                  </div>
                </>
              ) : (
                <div className="space-y-4">
                  <div>
                    <Label htmlFor="nombre">Nombre</Label>
                    <Input
                      id="nombre"
                      value={editData.nombre}
                      onChange={(e) => setEditData({ ...editData, nombre: e.target.value })}
                      className="mt-1"
                    />
                  </div>
                  <div>
                    <Label htmlFor="email">Email</Label>
                    <Input
                      id="email"
                      type="email"
                      value={editData.email}
                      onChange={(e) => setEditData({ ...editData, email: e.target.value })}
                      className="mt-1"
                    />
                  </div>
                  <div className="flex gap-2">
                    <Button onClick={handleUpdateProfile} className="bg-green-600 hover:bg-green-700">
                      <Save className="w-4 h-4 mr-2" />
                      Guardar
                    </Button>
                    <Button onClick={() => setEditMode(false)} variant="outline">
                      <X className="w-4 h-4 mr-2" />
                      Cancelar
                    </Button>
                  </div>
                </div>
              )}
              
              <Separator className="my-6" />
              
              {/* Estad√≠sticas mejoradas */}
              <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
                <div className="group relative overflow-hidden bg-gradient-to-br from-orange-50 to-orange-100 p-6 rounded-2xl border-2 border-orange-200 hover:shadow-xl transition-all duration-300">
                  <div className="absolute top-0 right-0 w-20 h-20 bg-orange-400/20 rounded-full -mr-10 -mt-10"></div>
                  <Trophy className="w-8 h-8 text-orange-600 mb-2" />
                  <p className="text-gray-600 text-sm mb-1">Total de Rese√±as</p>
                  <p className="text-4xl font-bold text-orange-600">{userData.totalReviews}</p>
                </div>
                
                <div className="group relative overflow-hidden bg-gradient-to-br from-yellow-50 to-yellow-100 p-6 rounded-2xl border-2 border-yellow-200 hover:shadow-xl transition-all duration-300">
                  <div className="absolute top-0 right-0 w-20 h-20 bg-yellow-400/20 rounded-full -mr-10 -mt-10"></div>
                  <Star className="w-8 h-8 text-yellow-600 mb-2" />
                  <p className="text-gray-600 text-sm mb-1">Valoraci√≥n Promedio</p>
                  <div className="flex items-center gap-2">
                    <p className="text-4xl font-bold text-yellow-600">{userData.avgRating.toFixed(1)}</p>
                    <Star className="w-6 h-6 fill-yellow-400 text-yellow-400" />
                  </div>
                </div>
                
                <div className="group relative overflow-hidden bg-gradient-to-br from-purple-50 to-purple-100 p-6 rounded-2xl border-2 border-purple-200 hover:shadow-xl transition-all duration-300">
                  <div className="absolute top-0 right-0 w-20 h-20 bg-purple-400/20 rounded-full -mr-10 -mt-10"></div>
                  <TrendingUp className="w-8 h-8 text-purple-600 mb-2" />
                  <p className="text-gray-600 text-sm mb-1">Lugares Visitados</p>
                  <p className="text-4xl font-bold text-purple-600">{userData.totalReviews}</p>
                </div>
              </div>
            </div>
          </div>
        </Card>
        
        {/* Secci√≥n de rese√±as */}
        <Card className="p-8 rounded-3xl shadow-2xl bg-white/80 backdrop-blur border-2 border-white">
          <div className="flex items-center gap-3 mb-6">
            <div className="p-3 bg-gradient-to-br from-blue-100 to-orange-100 rounded-xl">
              <Star className="w-6 h-6 text-blue-600" />
            </div>
            <h2 className="text-3xl font-bold text-gray-900">Mis Rese√±as</h2>
          </div>
          
          {reviews.length === 0 ? (
            <div className="text-center py-16 bg-gradient-to-br from-gray-50 to-blue-50 rounded-2xl">
              <div className="text-6xl mb-4">üìù</div>
              <p className="text-gray-600 text-lg mb-2">A√∫n no has publicado ninguna rese√±a</p>
              <p className="text-sm text-gray-400">¬°Comparte tu experiencia en tus lugares favoritos!</p>
            </div>
          ) : (
            <div className="space-y-4">
              {reviews.map((review) => (
                <Card 
                  key={review.reviewId} 
                  className="group p-6 rounded-2xl hover:shadow-xl transition-all duration-300 border-l-4 border-l-orange-400 bg-gradient-to-r from-white to-orange-50/30"
                >
                  {editingReview === review.reviewId ? (
                    <div className="space-y-4">
                      <div>
                        <Label>Valoraci√≥n</Label>
                        <div className="flex gap-2 mt-2">
                          {[1, 2, 3, 4, 5].map((rating) => (
                            <button
                              key={rating}
                              onClick={() => setEditReviewData({ ...editReviewData, puntuacion: rating })}
                            >
                              <Star
                                className={`w-8 h-8 transition-colors ${
                                  rating <= editReviewData.puntuacion
                                    ? 'fill-yellow-400 text-yellow-400'
                                    : 'text-gray-300'
                                }`}
                              />
                            </button>
                          ))}
                        </div>
                      </div>
                      <div>
                        <Label>Comentario</Label>
                        <Textarea
                          value={editReviewData.comentario}
                          onChange={(e) => setEditReviewData({ ...editReviewData, comentario: e.target.value })}
                          className="mt-2"
                          rows={4}
                        />
                      </div>
                      <div className="flex gap-2">
                        <Button 
                          onClick={() => handleSaveReview(review.reviewId)}
                          className="bg-green-600 hover:bg-green-700"
                        >
                          <Save className="w-4 h-4 mr-2" />
                          Guardar
                        </Button>
                        <Button onClick={() => setEditingReview(null)} variant="outline">
                          Cancelar
                        </Button>
                      </div>
                    </div>
                  ) : (
                    <>
                      <div className="flex items-start justify-between mb-3">
                        <div className="flex-1 cursor-pointer" onClick={() => onRestaurantClick(review.placeId)}>
                          <div className="flex items-center gap-2 mb-2">
                            <MapPin className="w-4 h-4 text-orange-600" />
                            <h3 className="text-xl font-semibold text-gray-900 hover:text-blue-600 transition-colors">
                              {review.restaurantName || 'Restaurante'}
                            </h3>
                          </div>
                          <div className="flex items-center gap-2">
                            <div className="flex gap-1">
                              {Array.from({ length: 5 }).map((_, i) => (
                                <Star
                                  key={i}
                                  className={`w-5 h-5 ${
                                    i < review.puntuacion
                                      ? 'fill-yellow-400 text-yellow-400'
                                      : 'text-gray-300'
                                  }`}
                                />
                              ))}
                            </div>
                            <Badge className="bg-orange-100 text-orange-700 border-0">
                              {review.puntuacion}/5
                            </Badge>
                          </div>
                        </div>
                        <span className="text-gray-500 text-sm">
                          {new Date(review.fecha).toLocaleDateString('es-ES')}
                        </span>
                      </div>
                      
                      <p className="text-gray-700 mb-4 leading-relaxed">{review.comentario}</p>
                      
                      <div className="flex gap-2">
                        <Button
                          onClick={() => handleEditReview(review)}
                          variant="outline"
                          size="sm"
                          className="rounded-lg hover:bg-blue-50 hover:border-blue-300 hover:text-blue-600"
                        >
                          <Edit className="w-4 h-4 mr-2" />
                          Editar
                        </Button>
                        <Button
                          onClick={() => handleDeleteReview(review.reviewId)}
                          variant="outline"
                          size="sm"
                          className="rounded-lg text-red-600 hover:bg-red-50 hover:border-red-300"
                        >
                          <Trash2 className="w-4 h-4 mr-2" />
                          Eliminar
                        </Button>
                      </div>
                    </>
                  )}
                </Card>
              ))}
            </div>
          )}
        </Card>
      </div>
    </div>
  );
}
